pyecharts的Geo模块的使用
您只需要知道数据类型data={"city1"：['name1','name2','name3'], 'city2':['name4','name5']}

#导入pyecharts的Geo模块

from pyecharts.charts import Geo

#全局配置

from pyecharts import options as opts

# 准备数据

data = { "北京": ["chicken", "周某"], "上海": ["乌鸡"], "广州": ["伟哥"], '西安':['杜天帝','wc']

# ... 添加更多城市和对应的人名

}

# 创建 Geo 对象

geo = Geo()

# 创建 Geo 对象

geo = Geo()

# 使用Geo()函数创建Geo对象,把背景颜色设置为"#475262"，并将结果赋值给变量geo

geo = Geo( init_opts=opts.InitOpts(bg_color="#475262") )

# 添加数据

geo.add_schema(maptype="china", itemstyle_opts=opts.ItemStyleOpts( color="#2d3948", border_color="#58667a" ), emphasis_itemstyle_opts=opts.ItemStyleOpts( color="#2a333d" ) )

# 选择中国地图

for city, names in data.items():

# 由于人数相同，这里我们使用一个固定的大小，或者您可以自定义一个大小

geo.add(

"高中同学分布",

[(city, len(names))],

# 使用名字列表的长度作为数据值

type_='scatter', symbol_size=len(names)*5,

# 设置一个固定的大小

label_opts=opts.LabelOpts(is_show=True, formatter=f"{city}"), tooltip_opts=opts.TooltipOpts(formatter=f"{city}: {', '.join(names)}") # 显示每个城市的人名列表 )

# 设置全局配置项

geo.set_global_opts( title_opts=opts.TitleOpts(title="高中同学分布图"), visualmap_opts=opts.VisualMapOpts(is_show=False),

# 不显示视觉映射组件

)

# 设置系列配置项，不显示每个城市的人数

geo.set_series_opts(label_opts=opts.LabelOpts(is_show=False))

# 渲染图表到文件,记得选一个自己的保存路径

geo.render('geo_chart.html')
